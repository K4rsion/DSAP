109 
110  StkFloat temp2 = 0.0;
111  for ( unsigned int i=0; i<nModes_; i++ )
112  temp2 += filters_[i]->tick(temp);
113 
114  temp2 -= temp2 * directGain_;
115  temp2 += directGain_ * temp;
