274  virtual bool begin() {
275  // setup spi
276  bool result = true;
277  for (auto &tmp : spi) {
278  result &= tmp.begin();
279  }
280  // setup i2c
281  for (auto &tmp : i2c) {
282  result &= tmp.begin();
283  }
284  // setup pins
285  for (auto &tmp : pins) {
286  if (tmp.pin != -1) {
287 
288  switch (tmp.function) {
289  case HEADPHONE_DETECT:
290  pinMode(tmp.pin, INPUT_PULLUP);
291  break;
292  case AUXIN_DETECT:
293  pinMode(tmp.pin, INPUT_PULLUP);
294  break;
295  case KEY:
296  pinMode(tmp.pin, INPUT);
297  break;
298  case LED:
299  pinMode(tmp.pin, OUTPUT);
300  break;
301  case PA:
302  pinMode(tmp.pin, OUTPUT);
303  digitalWrite(tmp.pin, HIGH);
304  break;
305  }
306  }
307  }
308  return result;
309  }
310 
311  void end() {
312  // setup spi
313  for (auto &tmp : spi) {
314  tmp.end();
315  }
316  // setup i2c
317  for (auto &tmp : i2c) {
318  tmp.end();
319  }
320  }
