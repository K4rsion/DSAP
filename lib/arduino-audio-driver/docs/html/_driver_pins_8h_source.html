419  void setupPinMode() {
420  // setup pins
421  for (auto &tmp : pins) {
422  if (tmp.pin != -1) {
423  if (!hasConflict(tmp.pin)) {
424  AD_LOGD("pinMode %d", tmp.pin);
425  switch (tmp.pin_logic) {
426  case PinLogic::InputActiveHigh:
427  pinMode(tmp.pin, INPUT);
428  break;
429  case PinLogic::InputActiveLow:
430  pinMode(tmp.pin, INPUT_PULLUP);
431  break;
432  case PinLogic::Input:
433  pinMode(tmp.pin, INPUT);
434  break;
435  case PinLogic::Output:
436  pinMode(tmp.pin, OUTPUT);
437  break;
438  default:
439  // do nothing
440  break;
441  }
442  } else {
443  AD_LOGW("Pin '%d' not set up because of conflict", tmp.pin);
444  tmp.active = false;
445  }
446  } else {
447  AD_LOGD("Pin is -1");
448  }
449  AD_LOGD("Pin %d set", tmp.pin);
450  }
451  }
